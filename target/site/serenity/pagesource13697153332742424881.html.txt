function delete_all() {

    var checkboxes = $("[class=checkboxcls]:checked");
    var table_name = "pt_blog_categories";
    var main_key = "cat_id";

    var all_data = [];
    $.each(checkboxes, function (index, object, container) {
        all_data.push($(object).val())
    });
    if (all_data.length != 0) {
        var answer = confirm("Are you sure you want to delete?");
        if (answer) {
            $.post("https://www.phptravels.net/admin/hotels/DeleteAll", {primery_keys: all_data,table_name : table_name,main_key:main_key}, function (theResponse) {
                location.reload();
            });

        } else {
            location.reload();
            return false;
        }
    } else {
        alert("Please at least select one item.")
    }
}
